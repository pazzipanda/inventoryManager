--create trigger for inserting account

CREATE OR REPLACE trigger NEWACCOUNT
	before insert on ACCOUNTS
	for each row
begin
	if :NEW.ACCOUNTID is null then
		select ACCOUNTID_SEQUENCE.nextval into :NEW.ACCOUNTID from dual;
	end if;
end;
/
